def adf_test(ts):
    res = adfuller(ts)
    print('ADF = %f' % res[0])
    print('p-value = %f' % res[1])
    print('Critical Values:')
    for k, v in res[4].items():
        print('\t%s: %.3f' % (k, v))
    return res

    print("Val Index:", val_index)

avg_rmse = []
for pdq in pdq_grid:
    print(pdq) # swap this out for tqdm
    rmse = []
    for train_index, val_index in tskfold.split(ts_train):
        
        # train-val split
        x_train = ts_train[train_index]
        y_val = ts_train[val_index]
        
        # log, s.diff
        x_d = x_train.diff(m)[m:] 
        
        # fit ARIMA
        model = ARIMA(x_d, order=pdq, enforce_stationarity=False, enforce_invertibility=False).fit(method_kwargs={'maxiter': 200})
        
        # forecast
        y_pred = model.forecast(h)
        
        # reverse log, s.diff
        y_pred = x_train[-m:].to_numpy() + y_pred.to_numpy()
        
        # compile error
        rmse.append(np.sqrt(mean_squared_error(y_val, y_pred)))
        
    # avg error
    avg_rmse.append(np.mean(rmse))
